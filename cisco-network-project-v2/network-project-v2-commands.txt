================================================================
 ROUTER 1 CONFIGURATION
 Model    : Cisco 2911
 Role     : RIP v2 — Gateway for HR (VLAN 10) and IT (VLAN 20)
 WAN R3   : GigabitEthernet0/1 = 10.0.0.1/30
 WAN R2   : GigabitEthernet0/2 = 10.0.3.1/30
 LAN HR   : GigabitEthernet0/0.10 = 192.168.10.1/24 (VLAN 10)
 LAN IT   : GigabitEthernet0/0.20 = 192.168.20.1/24 (VLAN 20)
================================================================

enable
configure terminal

hostname Router1

! WAN interface to Router3
interface GigabitEthernet0/1
 ip address 10.0.0.1 255.255.255.252
 no shutdown
 exit

! WAN interface to Router2 (direct link)
interface GigabitEthernet0/2
 ip address 10.0.3.1 255.255.255.252
 no shutdown
 exit

! Enable trunk LAN interface for Switch1
interface GigabitEthernet0/0
 no shutdown
 exit

! Sub-interface for HR VLAN 10
interface GigabitEthernet0/0.10
 encapsulation dot1Q 10
 ip address 192.168.10.1 255.255.255.0
 ip helper-address 192.168.50.10
 exit

! Sub-interface for IT VLAN 20
interface GigabitEthernet0/0.20
 encapsulation dot1Q 20
 ip address 192.168.20.1 255.255.255.0
 ip helper-address 192.168.50.10
 exit

! RIP v2 configuration
router rip
 version 2
 no auto-summary
 network 10.0.0.0
 network 192.168.10.0
 network 192.168.20.0
 exit

end
write memory


================================================================
 ROUTER 2 CONFIGURATION
 Model    : Cisco 2911
 Role     : RIP v2 + OSPF area 0 — REDISTRIBUTION BOUNDARY
            Gateway for Finance (VLAN 30) and Sales (VLAN 40)
 WAN R3   : GigabitEthernet0/1 = 10.0.1.1/30
 WAN R4   : GigabitEthernet0/2 = 10.0.2.1/30
 LAN Fin. : GigabitEthernet0/0.30 = 192.168.30.1/24 (VLAN 30)
 LAN Sales: GigabitEthernet0/0.40 = 192.168.40.1/24 (VLAN 40)
================================================================

enable
configure terminal

hostname Router2

! WAN interface to Router3
interface GigabitEthernet0/1
 ip address 10.0.1.1 255.255.255.252
 no shutdown
 exit

! WAN interface to Router4
interface GigabitEthernet0/2
 ip address 10.0.2.1 255.255.255.252
 no shutdown
 exit

! Enable trunk LAN interface for Switch2
interface GigabitEthernet0/0
 no shutdown
 exit

! Sub-interface for Finance VLAN 30
interface GigabitEthernet0/0.30
 encapsulation dot1Q 30
 ip address 192.168.30.1 255.255.255.0
 ip helper-address 192.168.50.10
 exit

! Sub-interface for Sales VLAN 40
interface GigabitEthernet0/0.40
 encapsulation dot1Q 40
 ip address 192.168.40.1 255.255.255.0
 ip helper-address 192.168.50.10
 exit

! RIP v2 configuration with OSPF redistribution
! "redistribute ospf 1 metric 2" injects OSPF routes (VLAN 50)
! into RIP so Router1 and Router3 learn about the Server Room.
router rip
 version 2
 no auto-summary
 network 10.0.0.0
 network 192.168.30.0
 network 192.168.40.0
 redistribute ospf 1 metric 2
 exit

! OSPF configuration with RIP redistribution
! "redistribute rip metric 20 subnets" injects RIP routes
! (VLAN 10,20,30,40,60) into OSPF so Router4 learns them.
! "subnets" is required to redistribute classless routes.
router ospf 1
 network 10.0.2.0 0.0.0.3 area 0
 redistribute rip metric 20 subnets
 exit

end
write memory


================================================================
 ROUTER 3 CONFIGURATION
 Model    : Cisco 2911
 Role     : RIP v2 + 1 static route — Transit + VLAN 60 gateway
 WAN R1   : GigabitEthernet0/0 = 10.0.0.2/30
 WAN R2   : GigabitEthernet0/1 = 10.0.1.2/30
 LAN Mgmt : GigabitEthernet0/2.60 = 192.168.60.1/24 (VLAN 60)
================================================================

enable
configure terminal

hostname Router3

! WAN interface to Router1
interface GigabitEthernet0/0
 ip address 10.0.0.2 255.255.255.252
 no shutdown
 exit

! WAN interface to Router2
interface GigabitEthernet0/1
 ip address 10.0.1.2 255.255.255.252
 no shutdown
 exit

! Enable trunk LAN interface for Switch4 (Management)
interface GigabitEthernet0/2
 no shutdown
 exit

! Sub-interface for Management VLAN 60
interface GigabitEthernet0/2.60
 encapsulation dot1Q 60
 ip address 192.168.60.1 255.255.255.0
 exit

! RIP v2 configuration
! "network 10.0.0.0" covers both WAN links (10.0.0.2 and 10.0.1.2)
! Router3 also advertises the Management VLAN (192.168.60.0)
router rip
 version 2
 no auto-summary
 network 10.0.0.0
 network 192.168.60.0
 exit

! Static route (demonstration)
! This explicitly routes traffic to the R2-R4 WAN link via Router2.
! Router3 also learns 10.0.2.0 via RIP from Router2, but the
! static route (AD=1) takes precedence over RIP (AD=120).
ip route 10.0.2.0 255.255.255.252 10.0.1.1

end
write memory


================================================================
 ROUTER 4 CONFIGURATION
 Model    : Cisco 2911
 Role     : OSPF area 0 + Default Route — Server Room gateway
 WAN R2   : GigabitEthernet0/1 = 10.0.2.2/30
 LAN Srv  : GigabitEthernet0/0.50 = 192.168.50.1/24 (VLAN 50)
================================================================

enable
configure terminal

hostname Router4

! WAN interface to Router2
interface GigabitEthernet0/1
 ip address 10.0.2.2 255.255.255.252
 no shutdown
 exit

! Enable trunk LAN interface for Switch3 (Server Room)
interface GigabitEthernet0/0
 no shutdown
 exit

! Sub-interface for Server Room VLAN 50
interface GigabitEthernet0/0.50
 encapsulation dot1Q 50
 ip address 192.168.50.1 255.255.255.0
 exit

! OSPF configuration
! Advertises the Server Room subnet and the WAN link to Router2.
! Router2 redistributes these OSPF routes into RIP.
router ospf 1
 network 192.168.50.0 0.0.0.255 area 0
 network 10.0.2.0 0.0.0.3 area 0
 exit

! Default route
! Any traffic for networks Router4 doesn't know specifically
! (e.g., VLAN 10, 20, 30, 40, 60) is sent to Router2 (10.0.2.1).
! Router2 knows all routes and will forward it correctly.
ip route 0.0.0.0 0.0.0.0 10.0.2.1

end
write memory

================================================================
 QUICK TROUBLESHOOTING
================================================================

Problem: PC gets no DHCP IP
  Check: ip helper-address on router sub-interface
  Check: DHCP pool configured on DHCP-Server
  Fix:   interface GigabitEthernet0/0.10
           ip helper-address 192.168.50.10

Problem: Ping between VLANs fails
  Check: show ip route on router
  Check: show ip interface brief (interfaces should be up/up)
  Check: VLAN created on switch (show vlan brief)

Problem: OSPF neighbor not forming (Router2 & Router4)
  Check: show ip ospf neighbor
  Check: both routers use area 0
  Check: no ACL blocking the OSPF multicast 224.0.0.5

Problem: Redistribution not working
  Check: show ip protocols on Router2
  Check: "redistribute ospf 1 metric 2" in RIP section
  Check: "redistribute rip metric 20 subnets" in OSPF section
         (the "subnets" keyword is required!)

Problem: Switch5 PCs get wrong VLAN
  Check: show interfaces trunk on Switch1 and Switch5
  Check: Fa0/24 on Switch1 is trunk
  Check: Fa0/1 on Switch5 is trunk
  Check: VLANs 10 and 20 are created on Switch5
