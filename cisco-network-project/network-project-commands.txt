 ROUTER 1 CONFIGURATION
 Model  : Cisco 2911
 Role   : Gateway for HR (VLAN 10) and IT (VLAN 20)
 WAN    : GigabitEthernet0/1 = 10.0.0.1/30 -> Router3
 LAN HR : GigabitEthernet0/0.10 = 192.168.10.1/24 -> Switch1
 LAN IT : GigabitEthernet0/2.20 = 192.168.20.1/24 -> Switch2
================================================================

! --- Enter privileged mode ---
enable

! --- Enter global configuration mode ---
configure terminal

! --- Set hostname ---
hostname Router1

! --- Configure WAN interface to Router3 ---
interface GigabitEthernet0/1
 ip address 10.0.0.1 255.255.255.252
 no shutdown
 exit

! --- Enable LAN interface for HR (Switch1) ---
interface GigabitEthernet0/0
 no shutdown
 exit

! --- Sub-interface for HR VLAN 10 ---
interface GigabitEthernet0/0.10
 encapsulation dot1Q 10
 ip address 192.168.10.1 255.255.255.0
 ip helper-address 192.168.50.10
 exit

! --- Enable LAN interface for IT (Switch2) ---
interface GigabitEthernet0/2
 no shutdown
 exit

! --- Sub-interface for IT VLAN 20 ---
interface GigabitEthernet0/2.20
 encapsulation dot1Q 20
 ip address 192.168.20.1 255.255.255.0
 ip helper-address 192.168.50.10
 exit

! --- Configure RIP version 2 ---
router rip
 version 2
 no auto-summary
 network 10.0.0.0
 network 192.168.10.0
 network 192.168.20.0
 exit

! --- Save configuration ---
end
write memory

 SECTION 14 — FIX ROUTING (RIP + STATIC + DEFAULT)
 Run these commands to restore full network connectivity.
 Fix Router1, Router5, and Router4 (in that order).
 Router2 and Router3 need no changes.
================================================================

! ============================================================
! FIX ROUTER1 — Restore RIP + Add Default Route
! ============================================================
enable
configure terminal

router rip
 version 2
 no auto-summary
 network 10.0.0.0
 network 192.168.10.0
 network 192.168.20.0
 exit

ip route 0.0.0.0 0.0.0.0 10.0.0.2

end
write memory


! ============================================================
! FIX ROUTER5 — Restore RIP + Add Static Routes for Router1 side
! ============================================================
enable
configure terminal

router rip
 version 2
 no auto-summary
 network 10.0.2.0
 network 10.0.3.0
 network 192.168.50.0
 redistribute static
 exit

ip route 192.168.10.0 255.255.255.0 10.0.2.1
ip route 192.168.20.0 255.255.255.0 10.0.2.1
ip route 10.0.0.0 255.255.255.252 10.0.2.1

end
write memory


! ============================================================
! FIX ROUTER4 — Clean up static routes + keep default + RIP
! ============================================================
enable
configure terminal

no ip route 192.168.30.0 255.255.255.0 10.0.1.1
no ip route 192.168.40.0 255.255.255.0 10.0.1.1

router rip
 version 2
 no auto-summary
 network 10.0.1.0
 network 10.0.3.0
 exit

! Keep the default route (sends unknown traffic to Router5 at 10.0.3.2)
! If it already exists, this command is safe to run again — it will just be ignored.
ip route 0.0.0.0 0.0.0.0 10.0.3.2

end
write memory
